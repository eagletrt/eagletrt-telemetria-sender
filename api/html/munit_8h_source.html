295  do { \

297  const double munit_tmp_b_ = (b); \
298  const double munit_tmp_diff_ = ((munit_tmp_a_ - munit_tmp_b_) < 0) ? \

334 #define munit_assert_memory_equal(size, a, b) \
335  do { \

337  const unsigned char* munit_tmp_b_ = (const unsigned char*) (b); \

339  if (MUNIT_UNLIKELY(memcmp(munit_tmp_a_, munit_tmp_b_, munit_tmp_size_)) != 0) { \

343  munit_errorf("assertion failed: memory %s == %s, at offset %" MUNIT_SIZE_MODIFIER "u", \
344  #a, #b, munit_tmp_pos_); \
345  break; \
346  } \
347  } \
